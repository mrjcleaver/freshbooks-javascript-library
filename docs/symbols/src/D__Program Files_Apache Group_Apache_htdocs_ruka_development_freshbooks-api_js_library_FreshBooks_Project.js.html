136 /**
137 * process XML string response from LIST server method
138 */	
139 FreshBooks_Project.prototype.internalListing = function(responseStatus,XMLObject)
140 {
141 	var result = Object();
142 	var resultProjects = Array();
143 	if(responseStatus){
144 		var projects = XMLObject.getElementsByTagName('project');
145 		if(projects.length > 0){
146 			resultProjects.length = projects.length;		
147 			for(var i = 0; i < projects.length; i++){
148 				var thisProject = new FreshBooks_Project();
149 				thisProject.internalLoadXML(projects[i]);
150 				resultProjects[i] = thisProject;
151 			}
152 		}
153 	}
154 	var projectsInfo = XMLObject.getElementsByTagName('projects')[0];
155 	result.page = projectsInfo.getAttribute('page');
156 	result.perPage = projectsInfo.getAttribute('per_page');
157 	result.pages = projectsInfo.getAttribute('pages');
158 	result.total = projectsInfo.getAttribute('total');
159 	result.rows = resultProjects;
160 	return result;
161 }
