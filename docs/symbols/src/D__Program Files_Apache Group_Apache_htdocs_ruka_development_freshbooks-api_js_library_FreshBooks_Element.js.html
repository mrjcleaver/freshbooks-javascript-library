 33 	return result;

 36 /**
 37 * get xml dom node friendly value
 38 */
 39 FreshBooks_Element.prototype.getXMLElementValue = function(XMLObject,element){
 40 	
 41 	if(!XMLObject)
 42 		return "";
 43 	var nodes=XMLObject.childNodes;
 44 	for (var i = 0; i < nodes.length; i++){
 45 		if(nodes[i].nodeName == element){
 46 			
 47 			if(nodes[i].firstChild){
 48 				var result = nodes[i].firstChild.nodeValue;
 49 				return result;
 50 			}
 51 			else
 52 				return "";
 53 		}
 54 	}
 55 	return "";
 56 }

 58 /**
 59 * get xml dom node 
 60 */
 61 FreshBooks_Element.prototype.getXMLNode = function(XMLObject,element){
 62 	
 63 	var nodes=XMLObject.childNodes;
 64 	for (var i = 0; i < nodes.length; i++){
 65 		if(nodes[i].nodeName == element){
 66 			return nodes[i];
 67 		}
 68 	}
 69 	return null;
 70 }

 89 /**
 90 * processes response xml
 91 */ 	
 92 FreshBooks_Element.prototype.processResponse = function(XMLObject){
 93 	if(!XMLObject){
 94 		this.lastError = "Connection error";
 95 		return false;
 96 	}
 97 	var response = XMLObject.getElementsByTagName("response");
 98 	if(response != null && response.length)
 99 		var result = XMLObject.getElementsByTagName("response")[0].getAttribute("status") == "ok";
100 	else{
101 		this.lastError = "Parse error";
102 		return false;
103 	}
104 	
105 	if(!result){
106 		this.lastError = XMLObject.getElementsByTagName("error")[0].firstChild.nodeValue;
107 	}
108 	return result;		
109 }
