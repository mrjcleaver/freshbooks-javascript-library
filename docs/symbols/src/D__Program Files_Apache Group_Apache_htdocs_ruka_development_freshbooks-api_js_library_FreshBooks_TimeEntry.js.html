143 {
144 	var result = Object();
145 	var resultTimeEntries = Array();
146 	if(responseStatus){
147 		var timeEntries = XMLObject.getElementsByTagName('time_entry');
148 		if(timeEntries.length > 0){
149 			resultTimeEntries.length = timeEntries.length;		
150 			for(var i = 0; i < timeEntries.length; i++){
151 				var thisTimeEntry = new FreshBooks_TimeEntry();
152 				thisTimeEntry.internalLoadXML(timeEntries[i]);
153 				resultTimeEntries[i] = thisTimeEntry;
154 			}
155 		}
156 	}
157 	var timeEntriesInfo = XMLObject.getElementsByTagName('time_entries')[0];
158 	result.page = timeEntriesInfo.getAttribute('page');
159 	result.perPage = timeEntriesInfo.getAttribute('per_page');
160 	result.pages = timeEntriesInfo.getAttribute('pages');
161 	result.total = timeEntriesInfo.getAttribute('total');
162 	result.rows = resultTimeEntries;
163 	return result;
