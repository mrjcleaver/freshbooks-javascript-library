149 /**
150 * load lines array from XML object
151 */ 	
152 FreshBooks_BaseInvoice.prototype.loadLines = function(XMLObject){
153 	this.lines = Array();
154 	
155 	var lines = this.getXMLNode(XMLObject,"lines");
156 	if(!lines){
157 		return;
158 	}
159 	
160 	var children=lines.childNodes;
161 	
162 	for (var i = 0; i < children.length; i++){
163 		var child = children[i];
164 		if(child.nodeName == "line"){
165 			var j = this.lines.length;
166 			this.lines.length = j + 1;
167 			var line = Object();
168 			line["name"] = this.getXMLElementValue(child,"name");
169 			line["description"] = this.getXMLElementValue(child,"description");
170 			line["unitCost"] = this.getXMLElementValue(child,"unit_cost");
171 			line["quantity"] = this.getXMLElementValue(child,"quantity");
172 			line["amount"] = this.getXMLElementValue(child,"amount");
173 			line["tax1Name"] = this.getXMLElementValue(child,"tax1_name");
174 			line["tax2Name"] = this.getXMLElementValue(child,"tax2_name");
175 			line["tax1Percent"] = this.getXMLElementValue(child,"tax1_percent");
176 			line["tax2Percent"] = this.getXMLElementValue(child,"tax2_percent");
177 			this.lines[j] = line;
178 		}
179 	}
180 }
