  8 function FreshBooks_HttpClient(url,token,proxyUrl) {
  9 	this.url = url;
 10 	this.token = token;
 11 	this.proxyUrl = proxyUrl;
 12 	// branch for native XMLHttpRequest object
 13 	if (window.XMLHttpRequest) {
 14 		this.request = new XMLHttpRequest();
 15 		this.isIE = false;
 16 	}
 17 	// branch for IE/Windows ActiveX version
 18 	else{ 
 19 		if (window.ActiveXObject) {
 20 			this.isIE = true;
 21 			this.request = new ActiveXObject("Microsoft.XMLHTTP");
 22 		}
 23 	}
 24 	
 25 /**
 26  * send request
 27  */	
 28 	this.send = function(content) {
 29 		var url = "";
 30 		if(this.proxyUrl)
 31 			url = this.proxyUrl + "?url=" + this.url;
 32 		else
 33 			url = this.url;
 34 		var post = "token=" + this.token + "&content=" + content;
 35 		this.request.open("POST", url, false, this.token);
 36 		this.request.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
 37 		this.request.setRequestHeader("Content-length", post.length);
 38     this.request.send(post);
 39 		if(this.request.status == 200)
 40 			return this.request.responseXML;
 41 		else
 42 			return false; 
 43 	};
 44 }
