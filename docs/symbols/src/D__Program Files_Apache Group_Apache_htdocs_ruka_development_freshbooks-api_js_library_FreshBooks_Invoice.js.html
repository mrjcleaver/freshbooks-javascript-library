106 /**
107 * process XML string response from LIST server method
108 */	
109 FreshBooks_Invoice.prototype.internalListing = function(responseStatus,XMLObject)
110 {
111 	var result = Object();
112 	var resultInvoices = Array();
113 	if(responseStatus){
114 		var invoices = XMLObject.getElementsByTagName('invoice');
115 		if(invoices.length > 0){
116 			resultInvoices.length = invoices.length;		
117 			for(var i = 0; i < invoices.length; i++){
118 				var thisInvoice = new FreshBooks_Invoice();
119 				thisInvoice.internalLoadXML(invoices[i]);
120 				resultInvoices[i] = thisInvoice;
121 			}
122 		}
123 	}
124 	var invoicesInfo = XMLObject.getElementsByTagName('invoices')[0];
125 	result.page = invoicesInfo.getAttribute('page');
126 	result.perPage = invoicesInfo.getAttribute('per_page');
127 	result.pages = invoicesInfo.getAttribute('pages');
128 	result.total = invoicesInfo.getAttribute('total');
129 	result.rows = resultInvoices;
130 	return result;
131 }
