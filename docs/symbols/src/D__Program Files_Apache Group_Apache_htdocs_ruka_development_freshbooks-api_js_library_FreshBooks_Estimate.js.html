 85 /**
 86 * process XML string response from LIST server method
 87 */	 	
 88 FreshBooks_Estimate.prototype.internalListing = function(responseStatus,XMLObject)
 89 {
 90 	var result = Object();
 91 	var resultEstimates = Array();
 92 	if(responseStatus){
 93 		var estimates = XMLObject.getElementsByTagName('estimate');
 94 		if(estimates.length > 0){
 95 			resultEstimates.length = estimates.length;		
 96 			for(var i = 0; i < estimates.length; i++){
 97 				var thisEstimate = new FreshBooks_Estimate();
 98 				thisEstimate.internalLoadXML(estimates[i]);
 99 				resultEstimates[i] = thisEstimate;
100 			}
101 		}
102 	}
103 	var estimatesInfo = XMLObject.getElementsByTagName('estimates')[0];
104 	result.page = estimatesInfo.getAttribute('page');
105 	result.perPage = estimatesInfo.getAttribute('per_page');
106 	result.pages = estimatesInfo.getAttribute('pages');
107 	result.total = estimatesInfo.getAttribute('total');
108 	result.rows = resultEstimates;
109 	return result;
110 }
