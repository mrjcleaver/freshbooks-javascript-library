213 /**
214 * process XML string response from LIST server method
215 */		
216 FreshBooks_Client.prototype.internalListing = function(responseStatus,XMLObject)
217 {
218 	var result = Object();
219 	var resultClients = Array();
220 	if(responseStatus){
221 		var clients = XMLObject.getElementsByTagName('client');
222 		if(clients.length > 0){
223 			resultClients.length = clients.length;		
224 			for(var i = 0; i < clients.length; i++){
225 				var thisClient = new FreshBooks_Client();
226 				thisClient.internalLoadXML(clients[i]);
227 				resultClients[i] = thisClient;
228 			}
229 		}
230 	}
231 	var clientsInfo = XMLObject.getElementsByTagName('clients')[0];
232 	result.page = clientsInfo.getAttribute('page');
233 	result.perPage = clientsInfo.getAttribute('per_page');
234 	result.pages = clientsInfo.getAttribute('pages');
235 	result.total = clientsInfo.getAttribute('total');
236 	result.rows = resultClients;
237 	return result;
238 }
