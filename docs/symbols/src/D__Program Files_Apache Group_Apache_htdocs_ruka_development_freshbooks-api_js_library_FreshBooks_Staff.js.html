166 * process XML string response from LIST server method

168 FreshBooks_Staff.prototype.internalListing = function(responseStatus,XMLObject)
169 {	
170 	var result = Object();
171 	var resultStaffMembers = Array();
172 	if(responseStatus){
173 		var staffMembers = XMLObject.getElementsByTagName('member');
174 		if(staffMembers.length > 0){
175 			resultStaffMembers.length = staffMembers.length;		
176 			for(var i = 0; i < staffMembers.length; i++){
177 				var thisStaff = new FreshBooks_Staff();
178 				thisStaff.internalLoadXML(staffMembers[i]);
179 				resultStaffMembers[i] = thisStaff;
180 			}
181 		}
182 	}
183 	var membersInfo = XMLObject.getElementsByTagName('staff_members')[0];
184 	result.page = membersInfo.getAttribute('page');
185 	result.perPage = membersInfo.getAttribute('per_page');
186 	result.pages = membersInfo.getAttribute('pages');
187 	result.total = membersInfo.getAttribute('total');
188 	result.rows = resultStaffMembers;
189 	return result;
190 }
